package ${config.controllerPackage};

import zlx.base.BaseController;
import org.springframework.stereotype.Controller;
import org.springframework.web.bind.annotation.RequestBody;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RequestMethod;
import org.springframework.web.bind.annotation.ResponseBody;
import zlx.aop.Pojo;
import ${config.pojoPackage}.${config.pojoName};

import java.util.Map;

@Controller
@RequestMapping("/api/${config.pojoName?lower_case}")
@Pojo(${config.pojoName}.class)
public class ${config.controllerName} extends BaseController<${config.pojoName}>
{
    /**
     * ${table.info}添加
     */
    @RequestMapping(value = "/add", method = RequestMethod.PUT)
    @ResponseBody
    public Map add(@RequestBody ${config.pojoName} ${table.name.field()})
    {
        try
        {
            return ${table.name.field()}Service.add(${table.name.field()});
        } catch (Exception e)
        {
            return error(e);
        }
    }

    /**
     * ${table.info}删除
     */
    @RequestMapping(value = "/delete", method = RequestMethod.DELETE)
    @ResponseBody
    public Map delete(@RequestBody Long[] ids)
    {
        try
        {
            ${table.name.field()}Service.delete(ids, ${table.name.field()}Mapper);
            return success(DELETE_OK);
        } catch (Exception e)
        {
            return error(e);
        }
    }

    /**
     * ${table.info}修改
     */
    @RequestMapping(value = "/update", method = RequestMethod.PUT)
    @ResponseBody
    public Map update(@RequestBody ${config.pojoName} ${table.name.field()})
    {
        try
        {
            ${table.name.field()}Service.update(${table.name.field()});
            return success(UPDATE_OK);
        } catch (Exception e)
        {
            return error(e);
        }
    }

    /**
     * 浏览${table.info}
     */
    @RequestMapping(value = "search", method = RequestMethod.PUT)
    @ResponseBody
    public Map search(@RequestBody ${config.pojoName} ${table.name.field()})
    {
        try
        {
            return ${table.name.field()}Service.search(${table.name.field()});
        } catch (Exception e)
        {
            return error(e);
        }
    }
}
